00080                 counter++;

00084             else if(bufDescTable[clockHand].pinCnt == 0 && bufDescTable[clockHand].dirty == true) {
00085                 flushFile(bufDescTable[clockHand].file);
00086                 frame = clockHand;

00089             // if it is not pinned and not dirty, allocate it straight away
00090             else if(bufDescTable[clockHand].pinCnt == 0 && bufDescTable[clockHand].dirty == false) {
00091                 hashTable->remove(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);

00094             }
00095         }
00096         // if frame not valid, can allocate it straight away
00097         else if(bufDescTable[clockHand].valid == false)

00099             frame = clockHand;
